Contents:

```python
import hashlib
import json
from pathlib import Path
import importlib.util


def _feature_list_hash(selected_features):
    payload = json.dumps(sorted(selected_features), sort_keys=True).encode("utf-8")
    return hashlib.sha256(payload).hexdigest()[:10]


def write_generated_feature_module(selected_features, output_dir, dataset_slug):
    """
    Render a standalone, importable .py file that computes exactly `selected_features`
    (plus the raw OHLCV columns) by calling Learn.features.add_feature_library and
    filtering the result. Returns (file_path: Path, content_hash: str).

    File name pattern: f"{dataset_slug}_{feature_hash}.py" so re-running feature selection
    with a different feature set produces a new, non-colliding file, while re-running with
    the same selection is idempotent (same hash, same filename).
    """
    output_dir = Path(output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)
    feature_hash = _feature_list_hash(selected_features)
    file_path = output_dir / f"{dataset_slug}_{feature_hash}.py"

    module_source = f'''"""
    Auto-generated by Learn.feature_codegen.write_generated_feature_module.
    Do not edit by hand — regenerate from the training notebook instead.
    dataset_slug: {dataset_slug}
    feature_hash: {feature_hash}
    """
    from Learn.features import add_feature_library

    SELECTED_FEATURES = {selected_features!r}

    _OHLCV_COLS = ["Time", "Open", "High", "Low", "Close", "Volume"]


    def compute_features(df, include_mtf=True, regime_params=None):
        df = add_feature_library(df, include_mtf=include_mtf, regime_params=regime_params)
        keep = [c for c in _OHLCV_COLS if c in df.columns]
        keep += [c for c in SELECTED_FEATURES if c in df.columns and c not in keep]
        return df[keep]
    '''
    file_path.write_text(module_source, encoding="utf-8")
    content_hash = hashlib.sha256(file_path.read_bytes()).hexdigest()
    return file_path, content_hash


def load_generated_feature_module(file_path, expected_hash=None):
    """
    Dynamically import a module produced by write_generated_feature_module and return its
    compute_features callable. If expected_hash is provided, verify it against the file's
    sha256 before importing and raise ValueError on mismatch (fail loudly, no silent
    fallback to a different feature function).
    """
    file_path = Path(file_path)
    if not file_path.exists():
        raise FileNotFoundError(f"Generated feature module not found: {file_path}")
    if expected_hash is not None:
        actual_hash = hashlib.sha256(file_path.read_bytes()).hexdigest()
        if actual_hash != expected_hash:
            raise ValueError(
                f"Generated feature module hash mismatch for {file_path}: "
                f"expected {expected_hash}, got {actual_hash}"
            )
    module_name = f"_generated_features_{file_path.stem}"
    spec = importlib.util.spec_from_file_location(module_name, file_path)
    module = importlib.util.module_from_spec(spec)
    spec.loader.exec_module(module)
    return module.compute_features
```

Notes for the implementer:
- Use an actual triple-quoted string built with `.format`/f-string carefully — the above is
  illustrative; when implementing, make sure the generated file's docstring/content doesn't
  break on feature names containing special characters (feature names are plain identifiers in
  this codebase, so this is low risk, but validate).
- `regime_params` is NOT baked into the generated module — it continues to be stored separately in
  the model pack (as it already is today) and passed as a call argument, since regime_params can
  legitimately vary at eval time (e.g. backtest notebook cell already prints "Using regime_params
  from model pack").
